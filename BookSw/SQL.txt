CREATE TABLE public.books (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  title text NOT NULL,
  owner text NOT NULL,
  owner_id uuid,
  rating double precision DEFAULT 0.0,
  swaps bigint DEFAULT 0,
  description text,
  is_available boolean DEFAULT true,
  is_for_rent boolean DEFAULT false,
  rental_price_per_day double precision DEFAULT 0.0,
  created_at timestamp with time zone NOT NULL DEFAULT timezone('utc'::text, now()),
  category text DEFAULT 'Fiction'::text,
  CONSTRAINT books_pkey PRIMARY KEY (id),
  CONSTRAINT books_owner_id_fkey FOREIGN KEY (owner_id) REFERENCES auth.users(id)
);
CREATE TABLE public.chat_requests (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  sender_id uuid NOT NULL,
  receiver_id uuid NOT NULL,
  book_id bigint NOT NULL,
  type text NOT NULL,
  status text DEFAULT 'pending'::text,
  created_at timestamp with time zone DEFAULT now(),
  CONSTRAINT chat_requests_pkey PRIMARY KEY (id),
  CONSTRAINT chat_requests_sender_id_fkey FOREIGN KEY (sender_id) REFERENCES auth.users(id),
  CONSTRAINT chat_requests_receiver_id_fkey FOREIGN KEY (receiver_id) REFERENCES auth.users(id),
  CONSTRAINT chat_requests_book_id_fkey FOREIGN KEY (book_id) REFERENCES public.books(id)
);
CREATE TABLE public.messages (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  chat_request_id bigint,
  sender_id uuid NOT NULL,
  content text NOT NULL,
  created_at timestamp with time zone DEFAULT now(),
  CONSTRAINT messages_pkey PRIMARY KEY (id),
  CONSTRAINT messages_chat_request_id_fkey FOREIGN KEY (chat_request_id) REFERENCES public.chat_requests(id),
  CONSTRAINT messages_sender_id_fkey FOREIGN KEY (sender_id) REFERENCES auth.users(id)
);
CREATE TABLE public.profiles (
  id uuid NOT NULL,
  full_name text NOT NULL,
  email text NOT NULL UNIQUE,
  phone text,
  address text,
  created_at timestamp with time zone NOT NULL DEFAULT timezone('utc'::text, now()),
  CONSTRAINT profiles_pkey PRIMARY KEY (id),
  CONSTRAINT profiles_id_fkey FOREIGN KEY (id) REFERENCES auth.users(id)
);
CREATE TABLE public.rentals (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  book_id bigint,
  renter_id uuid,
  start_date date NOT NULL,
  end_date date NOT NULL,
  total_price double precision,
  status text DEFAULT 'pending'::text,
  created_at timestamp with time zone NOT NULL DEFAULT timezone('utc'::text, now()),
  CONSTRAINT rentals_pkey PRIMARY KEY (id),
  CONSTRAINT rentals_book_id_fkey FOREIGN KEY (book_id) REFERENCES public.books(id),
  CONSTRAINT rentals_renter_id_fkey FOREIGN KEY (renter_id) REFERENCES auth.users(id)
);

-- ==========================================
-- 1. TABLES SETUP
-- ==========================================

-- Profiles table
CREATE TABLE IF NOT EXISTS public.profiles (
  id uuid REFERENCES auth.users NOT NULL PRIMARY KEY,
  full_name TEXT NOT NULL,
  email TEXT UNIQUE NOT NULL,
  phone TEXT,
  address TEXT,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- Books table (with category and image_url)
CREATE TABLE IF NOT EXISTS public.books (
  id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  title TEXT NOT NULL,
  owner TEXT NOT NULL,
  owner_id uuid REFERENCES auth.users,
  rating float8 DEFAULT 0.0,
  swaps int8 DEFAULT 0,
  description TEXT,
  category TEXT DEFAULT 'Fiction',
  is_available BOOLEAN DEFAULT true,
  is_for_rent BOOLEAN DEFAULT false,
  rental_price_per_day float8 DEFAULT 0.0,
  image_url TEXT,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- Rentals table
CREATE TABLE IF NOT EXISTS public.rentals (
  id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  book_id bigint REFERENCES books(id),
  renter_id uuid REFERENCES auth.users(id),
  start_date DATE NOT NULL,
  end_date DATE NOT NULL,
  total_price float8,
  status TEXT DEFAULT 'pending',
  created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- Chat Requests table
CREATE TABLE IF NOT EXISTS public.chat_requests (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    sender_id UUID REFERENCES auth.users(id) NOT NULL,
    receiver_id UUID REFERENCES auth.users(id) NOT NULL,
    book_id BIGINT REFERENCES books(id) NOT NULL,
    type TEXT NOT NULL,
    status TEXT DEFAULT 'pending',
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Messages table
CREATE TABLE IF NOT EXISTS public.messages (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    chat_request_id BIGINT REFERENCES chat_requests(id) ON DELETE CASCADE,
    sender_id UUID REFERENCES auth.users(id) NOT NULL,
    content TEXT NOT NULL,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- ==========================================
-- 3. SECURITY POLICIES (With DROP IF EXISTS)
-- ==========================================

-- Profiles Policies
DROP POLICY IF EXISTS "Public profiles are viewable by everyone" ON public.profiles;
CREATE POLICY "Public profiles are viewable by everyone" ON public.profiles FOR SELECT USING (true);

DROP POLICY IF EXISTS "Users can update own profile" ON public.profiles;
CREATE POLICY "Users can update own profile" ON public.profiles FOR UPDATE USING (auth.uid() = id);

-- Books Policies
DROP POLICY IF EXISTS "Books are viewable by everyone" ON public.books;
CREATE POLICY "Books are viewable by everyone" ON public.books FOR SELECT USING (true);

DROP POLICY IF EXISTS "Users can insert their own books" ON public.books;
CREATE POLICY "Users can insert their own books" ON public.books FOR INSERT WITH CHECK (auth.uid() = owner_id);

DROP POLICY IF EXISTS "Users can update their own books" ON public.books;
CREATE POLICY "Users can update their own books" ON public.books FOR UPDATE USING (auth.uid() = owner_id);

-- Rentals Policies
DROP POLICY IF EXISTS "Users can view their own rentals" ON public.rentals;
CREATE POLICY "Users can view their own rentals" ON public.rentals FOR SELECT USING (auth.uid() = renter_id);

-- Chat Requests Policies
DROP POLICY IF EXISTS "Users can view their own chat requests" ON public.chat_requests;
CREATE POLICY "Users can view their own chat requests" ON public.chat_requests 
    FOR SELECT USING (auth.uid() = sender_id OR auth.uid() = receiver_id);

DROP POLICY IF EXISTS "Users can insert chat requests" ON public.chat_requests;
CREATE POLICY "Users can insert chat requests" ON public.chat_requests 
    FOR INSERT WITH CHECK (auth.uid() = sender_id);

DROP POLICY IF EXISTS "Users can update request status" ON public.chat_requests;
CREATE POLICY "Users can update request status" ON public.chat_requests 
    FOR UPDATE USING (auth.uid() = receiver_id);

-- Messages Policies
DROP POLICY IF EXISTS "Users can view messages in their chats" ON public.messages;
CREATE POLICY "Users can view messages in their chats" ON public.messages FOR SELECT USING (
    EXISTS (
        SELECT 1 FROM chat_requests 
        WHERE chat_requests.id = messages.chat_request_id 
        AND (chat_requests.sender_id = auth.uid() OR chat_requests.receiver_id = auth.uid())
    )
);

DROP POLICY IF EXISTS "Users can send messages" ON public.messages;
CREATE POLICY "Users can send messages" ON public.messages FOR INSERT WITH CHECK (auth.uid() = sender_id);

-- ==========================================
-- 4. TRIGGERS
-- ==========================================

CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS trigger AS $$
BEGIN
  INSERT INTO public.profiles (id, full_name, email, phone, address)
  VALUES (
    new.id, 
    new.raw_user_meta_data->>'full_name', 
    new.email,
    new.raw_user_meta_data->>'phone',
    new.raw_user_meta_data->>'address'
  );
  RETURN new;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users;
CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE PROCEDURE public.handle_new_user();

    bucket_id = 'book-images' AND auth.role() = 'authenticated'
);
